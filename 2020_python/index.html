dunder Methods

  • __init__

class Example:
	def __init__(self, numValue):
		self.numValue = numValue

	def __str__(self):
		return 'The number is ' + str(self.numValue)

	def __repr__(self):
		return '{numValue: ' + str(self.numValue) + '}'

ex = Example(3)

print(str(ex))
print(repr(ex))
>python3 dunder_init_str_repr.py

Privacy is a Lie

class Secretive:
	def __init__(self):
		self.__secret = 'loose lips sink ships'

	def __secretiveWork(self):
		print('make a stealth check')

inst = Secretive()

print(inst._Secretive__secret)
print(inst._Secretive__secretiveWork())

Everything is Dynamic

from types import MethodType

def attachable(self):
	print('attachable() thinks the fact is ' + self.fact)

myType = type("dynatype",(),{'fact':"bears can't lie"})
inst = myType()

print(type(inst))

myType.myBoundMethod = MethodType(attachable, myType)

inst.myBoundMethod()
python3 dynamic.py

attachable() thinks the fact is bears can't lie
